    ) -> Assets:
        """Batcher fee.

        For Minswap, the batcher fee decreases linearly from 2.0 ADA to 1.5 ADA as the

        if extra_assets is not None and MIN in extra_assets:

        else:
            fee_reduction = 0
        return self._batcher - Assets(lovelace=fee_reduction)
